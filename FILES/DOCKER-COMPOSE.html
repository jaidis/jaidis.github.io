# Adguard
---
version: "3.0"
services:
  adguardhome:
    image: adguard/adguardhome:armhf-latest
    container_name: adguardhome
    network_mode: bridge
    ports:
      - 53:53/tcp
      - 53:53/udp
      - 67:67/tcp
      - 67:67/udp
      - 853:853/tcp
      - 3000:3000/tcp
    volumes:
      - /home/pi/docker-config/adguardhome:/opt/adguardhome/conf
      - /home/pi/docker-data/adguardhome:/opt/adguardhome/work
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Europe/Madrid
    restart: unless-stopped
